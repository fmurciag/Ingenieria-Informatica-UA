CREATE TABLE deportista(nif char(9), nombre varchar2(40),fechanac date, categoria varchar2(8),
constraint pk_deportista primary key(nif));

create table campeonato(codigo char(20), nombre varchar2(60),
constraint pk_campeonato primary key(codigo));

create table PRUEBA(codigo char(10), fecha date, lugar varchar2(30), localidad varchar2(20), provincia varchar2(15), 
CAMPEONATO CHAR(20) CONSTRAINT FK_PRUEBA_CAMPEONATO REFERENCES CAMPEONATO, presupuesto number(7),
CONSTraint pk_prueba primary key(codigo));

create table PARTICIPAR(DEPORTISTA CHAR(9) CONSTRAINT FK_PARTICIPAR_DEPORTISTA REFERENCES DEPORTISTA,
PRUEBA CHAR(10) CONSTRAINT FK_PARTICIPAR_PRUEBA REFERENCES PRUEBA, PUNTOS NUMBER(3),
CONSTRAINT PK_PARTICIPAR PRIMARY KEY(DEPORTISTA,PRUEBA));

create table clasificacion(nif_deportista char(9), nombre_deportista varchar2(40),
campeonato char(20), categoria varchar2(8),posicion number(4),
constraint pk_clasificacion primary key(campeonato,categoria,posicion));


create table ganadores(campeonato char(20), categoria varchar2(8), nif_deportista char(9), nombre_deportista varchar2(40), puntos number(3),
 primary key( campeonato, categoria));

create table destacados(prueba char(10), categoria varchar2(8), nif_deportista char(9), nombre_deportista varchar2(40), puntos number(3),
 primary key( prueba, nif_deportista));
